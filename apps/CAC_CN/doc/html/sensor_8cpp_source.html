69statusErrDef sensor::initSensor() {
70 statusErrDef res = noError;
71
72 #if (TARGET_SYSTEM == _WIN32_)
73 #else
74 for (i = 0; i < MAX_SENSORS; i++)
75 {
76 switch (getSensorType(i + 1))
77 {
78 case 1: // 1 if the sensor is one of the MCP3008 channels
79 //copy the activation from the 'activation.csv' file
80 tabSensorActivated[i] = getActivation(i + nbValuesCN_Out_ByCN + nbValuesCN_Out / 2 + 2);
81 break;
82 case 2: // 2 if the sensor is connected the the rs485 modbus serial link
83 //create a modbus serial instance for this sensor
84 ctx[i] = modbus_new_rtu(MODBUS_DEVICELOC, getModbusBaudRate(i+1),
85 getModbusParity(i+1), getModbusDataBits(i+1), getModbusStopBit(i+1));
86 if (!ctx[i]) {
87 fprintf(stderr, "Failed to create the context: %s\n", modbus_strerror(errno));
88 }
89 if (modbus_connect(ctx[i]) == -1) {
90 fprintf(stderr, "Unable to connect: %s\n", modbus_strerror(errno));
91 modbus_free(ctx[i]);
92 }
93 //set the address of the slave (the sensor) to the modbus instance
94 modbus_set_slave(ctx[i], getModbusAddrRemoteSlave(i+1));
95 //test if we can read from the sensor (if a sensor is connected) if not we disable the reading
96 //in readChannels to avoid slowing down the program
97 num = modbus_read_registers(ctx[i], getModbusStartAddress(i + 1), MODBUS_NBREG, modbusReg);
98 if (num != 1)// number of read registers is not the one expected
99 {
100 fprintf(stderr, "Failed to read: %s\n", modbus_strerror(errno));
101 rs485Connected[i] = false;
102 }
103 else
104 {
105 printf("Modbus read registers success! \n");
106 valSensors[i] = modbusReg[0];
107 rs485Connected[i] = true;
108 }
109 break;
110 default:
111 break;
112 }
113 }
114 //test if we can read channels correctly
115 res = readChannels();
116 if (res != noError)
117 return res;
118 #endif
119 return res;
120}

170statusErrDef readChannels()
171{
172 statusErrDef res = noError;
173 memset(fd, 0, sizeof(fd));
174 memset(valSensors, 0, sizeof(valSensors));
175
176 //we open the sysfs files of the MCP3008 channels
177 for (i = 0; i < MAX_ADC; i++) {
178 if (tabSensorActivated[i]) {
179 fd[i] = openAdc(i);
180 if (fd[i] == ADC_READ_ERROR)
181 {
182 res = errOpenAdc;
183 break;
184 }
185 }
186 }
187
188 //we read the values inside those files
189 for (i = 0; i < MAX_ADC; i++) {
190 if (!tabSensorActivated[i])
191 continue;
192
193 valSensors[i] = readAdc(fd[i]);
194
195 // reset for next read
196 lseek(fd[i], 0, SEEK_SET);
197
198 if (valSensors[i] == ADC_READ_ERROR)
199 {
200 res = errReadAdc;
201 break;
202 }
203 }
204
205 //we close them for the next read
206 closeAdc();
207 //we read read the modbus sensors if they are connected
208 #if (TARGET_SYSTEM == _WIN32_)
209 #else
210 for (i = 0; i < MAX_SENSORS; i++)
211 {
212 if (getSensorType(i + 1) == 2 && rs485Connected[i])
213 {
214 num = modbus_read_registers(ctx[i], getModbusStartAddress(i+1), MODBUS_NBREG, modbusReg);
215 if (num != 1)// number of read registers is not the one expected
216 {
217 fprintf(stderr, "Failed to read: %s\n", modbus_strerror(errno));
218 }
219 else
220 {
221 printf("Modbus read registers success! \n");
222 valSensors[i] = modbusReg[0];
223 }
224 }
225 }
226
227 #endif
228 return res;
229}
