73statusErrDef valve::initValve()
74{
75 statusErrDef res = noError;
76
77 if (strcmp(CHIP_PATH, "") == 0 || strcmp(CHIP_PATH, " ") == 0)
78 {
79 perror("Error: GPIO chip path is not set.");
80 return errGPIOPathEmpty;
81 }
82
83 chip = gpiod_chip_open(CHIP_PATH);
84 if (!chip)
85 {
86 perror("gpiod_chip_open");
87 return errOpenGPIO;
88 }
89
90 for (int i = 0; i < MAX_VALVES; i++)
91 {
92 if (getActivation(i + nbValuesCN_Out_ByCN + 2))
93 {
94 offsets[i] = getPortGPIO(i + 1);
95 values[i] = getEtatInitialVannes(i + 1);
96
97 lines[i] = gpiod_chip_get_line(chip, offsets[i]);
98 if (!lines[i])
99 {
100 perror("gpiod_chip_get_lines");
101 gpiod_line_release(lines[i]);
102 gpiod_chip_close(chip);
103 return errGPIOGetLine;
104 }
105
106 // Request the line for output
107 if (gpiod_line_request_output(lines[i], "opl", 0) < 0) {
108 perror("Request line as output failed");
109 gpiod_line_release(lines[i]);
110 gpiod_chip_close(chip);
111 return errGPIORequestOutput;
112 }
113 }
114 }
115
116 res = getValvesInitValue();
117 if (res != noError)
118 return res;
119 res = actionnementValvesInit();
120 if (res != noError)
121 return res;
122
123 return res;
124}

226statusErrDef valve::verifDependanceValves()
227{
228 statusErrDef res = noError;
229 for (int i = 0; i < MAX_VALVES; i++)
230 {
231 if (getActivation(i + nbValuesCN_Out_ByCN + 2))
232 {
233 //if mode is automatic then we compare the valve dependances
234 //with the dependances in a general state CSV file
235 //if mode is manual then we compare and set the value of the gpio lines
236 //with the values taken from the MN through the OpenPOWERLINK input array.
237 switch (mode)
238 {
239 case automatic:
240 res = isDependanceActive(i + nbValuesCN_In_ByCN + 2);
241 switch (res)
242 {
243 //when all the dependances are activated
244 case noError:
245 printf("\n==========\n");
246 printf("ligne:%d\n", i + nbValuesCN_In_ByCN + 2);
247 printf("BEFORE : getValeur(i + nbValuesCN_In_ByCN + 2):%d , gpiod_line_get_value(lines[i]):%d\n",
248 getValeur(i + nbValuesCN_In_ByCN + 2), gpiod_line_get_value(lines[i]));
249 //if the timer is not started then we start it
250 if (!timerStarted[i])
251 {
252 printf("start timer %d !\n", i);
253 startTimerDependance(i);
254 }
255 //if the timer has been exceeded we activate the valve
256 if (isTimerExeeded(i) == 0)
257 {
258 printf("actionnement valve %d !\n", i);
259 res = actionnementValve(i);
260 if (res != noError)
261 return res;
262 printf("i:%d\n", i);
263 printf("AFTER : getValeur(i + nbValuesCN_In_ByCN + 2):%d , gpiod_line_get_value(lines[i]):%d\n",
264 getValeur(i + nbValuesCN_In_ByCN + 2), gpiod_line_get_value(lines[i]));
265 }
266 break;
267 case infoNoDepend:
268 break;
269 case infoValveAlreadyActivated:
270 break;
271 case infoAllDependNotActivated:
272 break;
273 default:
274 return res;
275 break;
276 }
277 break;
278 case manual:
279 if (gpiod_line_get_value(lines[i]) < 0)
280 return errGPIOGetValue;
281 else if (getValues_In_CN(i + nbValuesCN_In_ByCN + 1) == gpiod_line_get_value(lines[i]))
282 res = infoValveAlreadyActivated;
283 else
284 {
285 res = actionnementValve(i);
286 if (res != noError)
287 return res;
288 }
289 break;
290 default:
291 break;
292 }
293 }
294
295 }
296
297 return res;
298}

319statusErrDef valve::isDependanceActive(int ligne)
320{
321 statusErrDef res = noError;
322
323 int* tab = getDependanceVannes(ligne);
324 int cmpt = 0;
325
326 if (tab != NULL)
327 {
328 //compare the real value of a gpio line with the valve value associated with it in the general state CSV file.
329 if (ligne > nbValuesCN_In_ByCN && ligne < nbValuesCN_In_ByCN + nbValuesCN_In)
330 {
331 //check if we can read the gpio line value before comparing it
332 if (gpiod_line_get_value(lines[(ligne - 2) % (nbValuesCN_In + 1)]) < 0)
333 {
334 free(tab);
335 return errGPIOGetValue;
336 }
337 else if (getValeur(ligne) == gpiod_line_get_value(lines[(ligne - 2) % (nbValuesCN_In + 1)]))
338 {
339 free(tab);
340 return infoValveAlreadyActivated;
341 }
342 }
343
344 //count the number of valve dependances for a valve
345 do {
346 cmpt++;
347 } while (cmpt <= MAX_VALVES && tab[cmpt] != 0);
348
349 if (tab[0] != 0)
350 {
351 //compare the real value of all gpio lines with the valve values in the general state CSV file.
352 for (int i = 0; i < cmpt; i++)
353 {
354 //check if we can read the gpio line value before comparing it
355 if (tab[i] > nbValuesCN_In_ByCN && tab[i] < nbValuesCN_In_ByCN + nbValuesCN_In)
356 {
357 if (gpiod_line_get_value(lines[(tab[i] - 1) % (nbValuesCN_In + 1)]) < 0)
358 {
359 free(tab);
360 return errGPIODependValue;
361 }
362 else if (getValeur(tab[i] + 1) != gpiod_line_get_value(lines[(tab[i] - 1) % (nbValuesCN_In + 1)]))
363 {
364 free(tab);
365 return infoAllDependNotActivated;
366 }
367 }
368 else
369 return errDependOutsideOfRange;
370 }
371 }
372
373 printf("Dependances toutes activees ! \n");
374 }
375 else
376 res = infoNoDepend;
377
378 free(tab);
379
380 return res;
381}
