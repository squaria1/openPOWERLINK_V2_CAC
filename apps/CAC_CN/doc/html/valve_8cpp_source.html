55statusErrDef valve::initValve()
56{
57 statusErrDef res = noError;
58
59 if (CHIP_PATH == "" || CHIP_PATH == " ")
60 {
61 perror("Error: GPIO chip path is not set.");
62 return errGPIOPathEmpty;
63 }
64
65 chip = gpiod_chip_open(CHIP_PATH);
66 if (!chip)
67 {
68 perror("gpiod_chip_open");
69 return errOpenGPIO;
70 }
71
72 for (int i = 0; i < MAX_VALVES; i++)
73 {
74 if (getActivation(i + nbValuesCN_Out_ByCN + 2))
75 {
76 offsets[i] = getPortGPIO(i + 1);
77 values[i] = getEtatInitialVannes(i + 1);
78
79 lines[i] = gpiod_chip_get_line(chip, offsets[i]);
80 if (!lines[i])
81 {
82 perror("gpiod_chip_get_lines");
83 gpiod_line_release(lines[i]);
84 gpiod_chip_close(chip);
85 return errGPIOGetLine;
86 }
87
88 // Request the line for output
89 if (gpiod_line_request_output(lines[i], "opl", 0) < 0) {
90 perror("Request line as output failed");
91 gpiod_line_release(lines[i]);
92 gpiod_chip_close(chip);
93 return errGPIORequestOutput;
94 }
95 }
96 }
97
98 res = getValvesInitValue();
99 if (res != noError)
100 return res;
101 res = actionnementValvesInit();
102 if (res != noError)
103 return res;
104
105 return res;
106}

208statusErrDef valve::verifDependanceValves()
209{
210 statusErrDef res = noError;
211 for (int i = 0; i < MAX_VALVES; i++)
212 {
213 if (getActivation(i + nbValuesCN_Out_ByCN + 2))
214 {
215 //if mode is automatic then we compare the valve dependances
216 //with the dependances in a general state CSV file
217 //if mode is manual then we compare and set the value of the gpio lines
218 //with the values taken from the MN through the OpenPOWERLINK input array.
219 switch (mode)
220 {
221 case automatic:
222 res = isDependanceActive(i + nbValuesCN_In_ByCN + 2);
223 switch (res)
224 {
225 //when all the dependances are activated
226 case noError:
227 printf("\n==========\n");
228 printf("ligne:%d\n", i + nbValuesCN_In_ByCN + 2);
229 printf("BEFORE : getValeur(i + nbValuesCN_In_ByCN + 2):%d , gpiod_line_get_value(lines[i]):%d\n",
230 getValeur(i + nbValuesCN_In_ByCN + 2), gpiod_line_get_value(lines[i]));
231 //if the timer is not started then we start it
232 if (!timerStarted[i])
233 {
234 printf("start timer %d !\n", i);
235 startTimerDependance(i);
236 }
237 //if the timer has been exceeded we activate the valve
238 if (isTimerExeeded(i) == 0)
239 {
240 printf("actionnement valve %d !\n", i);
241 res = actionnementValve(i);
242 if (res != noError)
243 return res;
244 printf("i:%d\n", i);
245 printf("AFTER : getValeur(i + nbValuesCN_In_ByCN + 2):%d , gpiod_line_get_value(lines[i]):%d\n",
246 getValeur(i + nbValuesCN_In_ByCN + 2), gpiod_line_get_value(lines[i]));
247 }
248 break;
249 case infoNoDepend:
250 break;
251 case infoValveAlreadyActivated:
252 break;
253 case infoAllDependNotActivated:
254 break;
255 default:
256 return res;
257 break;
258 }
259 break;
260 case manual:
261 if (gpiod_line_get_value(lines[i]) < 0)
262 return errGPIOGetValue;
263 else if (getValues_In_CN(i + nbValuesCN_In_ByCN + 1) == gpiod_line_get_value(lines[i]))
264 res = infoValveAlreadyActivated;
265 else
266 {
267 res = actionnementValve(i);
268 if (res != noError)
269 return res;
270 }
271 break;
272 default:
273 break;
274 }
275 }
276
277 }
278
279 return res;
280}

301statusErrDef valve::isDependanceActive(int ligne)
302{
303 statusErrDef res = noError;
304
305 int* tab = getDependanceVannes(ligne);
306 int cmpt = 0;
307
308 if (tab != NULL)
309 {
310 //compare the real value of a gpio line with the valve value associated with it in the general state CSV file.
311 if (ligne > nbValuesCN_In_ByCN && ligne < nbValuesCN_In_ByCN + nbValuesCN_In)
312 {
313 //check if we can read the gpio line value before comparing it
314 if (gpiod_line_get_value(lines[(ligne - 2) % (nbValuesCN_In + 1)]) < 0)
315 {
316 free(tab);
317 return errGPIOGetValue;
318 }
319 else if (getValeur(ligne) == gpiod_line_get_value(lines[(ligne - 2) % (nbValuesCN_In + 1)]))
320 {
321 free(tab);
322 return infoValveAlreadyActivated;
323 }
324 }
325
326 //count the number of valve dependances for a valve
327 do {
328 cmpt++;
329 } while (cmpt <= MAX_VALVES && tab[cmpt] != 0);
330
331 if (tab[0] != 0)
332 {
333 //compare the real value of all gpio lines with the valve values in the general state CSV file.
334 for (int i = 0; i < cmpt; i++)
335 {
336 //check if we can read the gpio line value before comparing it
337 if (tab[i] > nbValuesCN_In_ByCN && tab[i] < nbValuesCN_In_ByCN + nbValuesCN_In)
338 {
339 if (gpiod_line_get_value(lines[(tab[i] - 1) % (nbValuesCN_In + 1)]) < 0)
340 {
341 free(tab);
342 return errGPIODependValue;
343 }
344 else if (getValeur(tab[i] + 1) != gpiod_line_get_value(lines[(tab[i] - 1) % (nbValuesCN_In + 1)]))
345 {
346 free(tab);
347 return infoAllDependNotActivated;
348 }
349 }
350 else
351 return errDependOutsideOfRange;
352 }
353 }
354
355 printf("Dependances toutes activees ! \n");
356 }
357 else
358 res = infoNoDepend;
359
360 free(tab);
361
362 return res;
363}
