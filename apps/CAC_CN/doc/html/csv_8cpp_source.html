357void remplirStructure(char* ligne, int id)
358{
359 char* token = strtok(ligne, ";");
360 int colonne = 0;
361
362 while (token != NULL)
363 {
364 if (colonne == 2)
365 {
366 if (atoi(token) == 0)
367 dataEtats->valeur[id] = NULL;
368 else
369 dataEtats->valeur[id] = atoi(token);
370 }
371 else if (colonne == 4)
372 {
373 char* token2 = strtok(token, "|");
374 int i = 0;
375 while (token2 != NULL)
376 {
377 int valeurDependance = atoi(token2);
378 dataEtats->dependanceVannes[id][i] = valeurDependance;
379 i++;
380 token2 = strtok(NULL, "|");
381 }
382 }
383 else if (colonne == 3)
384 dataEtats->timerVannes[id] = atof(token);
385 colonne++;
386 token = strtok(NULL, ";");
387 }
388}
