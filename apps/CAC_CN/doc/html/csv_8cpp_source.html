372void remplirStructure(char* ligne, int id)
373{
374 char* token = strtok(ligne, ";");
375 char* token2 = NULL;
376 int colonne = 0, i = 0, valeurDependance = 0;
377
378 while (token != NULL)
379 {
380 switch (colonne)
381 {
382 case 2:
383 if (atoi(token) == 0)
384 dataEtats->valeur[id] = 0;
385 else
386 dataEtats->valeur[id] = atoi(token);
387 break;
388 case 3:
389 dataEtats->timerVannes[id] = atof(token);
390 break;
391 case 4:
392 token2 = strtok(token, "|");
393 i = 0;
394 while (token2 != NULL)
395 {
396 valeurDependance = atoi(token2);
397 dataEtats->dependanceVannes[id][i] = valeurDependance;
398 i++;
399 token2 = strtok(NULL, "|");
400 }
401 break;
402 default:
403 break;
404 }
405 colonne++;
406 token = strtok(NULL, ";");
407 }
408}
