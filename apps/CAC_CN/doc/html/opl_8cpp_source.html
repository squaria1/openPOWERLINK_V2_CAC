518tOplkError processSync()
519{
520 tOplkError res = kErrorOk;
521
522 if (oplk_waitSyncEvent(100000) != kErrorOk)
523 return res;
524
525
526 res = oplk_exchangeProcessImageOut();
527 if (res != kErrorOk)
528 return res;
529
530
531 //Process PI_IN --> variables entrant dans le CN
532 values_In_CN_l[nbValuesCN_In_ByCN] = pProcessImageIn_l->in_CN_array[nbValuesCN_In_ByCN];
533
534 int skipSensorsOutFromIn = 1;
535 // Example : CN3 and 3 CNs --> from nbValuesCN_Out_ByCN = 75 / 3 * (3 - 1) = 50 to nbValuesCN_Out_ByCN + nbValuesCN_Out = 50 + 25 = 75
536 switch (mode)
537 {
538 case automatic: // mode automatique : lecture de l'état des vannes depuis le CSV de l'etat general actuel
539 break;
540 case manual: // mode manuel : l'état des vannes proviennent directement du MN
541 for (int i = 1; i < SIZE_IN; i++)
542 {
543 if (i % (nbValuesCN_In + 1) == 0)
544 skipSensorsOutFromIn += nbValuesCN_In + 1;
545 else
546 skipSensorsOutFromIn += 1;
547 if (i % (nbValuesCN_In + 1) != 0 && activated_Out_CN_l[skipSensorsOutFromIn])
548 values_In_CN_l[i] = pProcessImageIn_l->in_CN_array[i];
549 }
550 break;
551 default:
552 break;
553 }
554
555 //Process PI_OUT --> variables sortant du CN
556 setValues_Out_CN();
557
558 for (int i = nbValuesCN_Out_ByCN; i < nbValuesCN_Out_ByCN + nbValuesCN_Out + 1; i++)
559 {
560 if (activated_Out_CN_l[i + 1])
561 pProcessImageOut_l->out_CN_array[i] = values_Out_CN_l[i];
562 }
563
564 res = oplk_exchangeProcessImageIn();
565 if (res != kErrorOk)
566 return res;
567
568 return res;
569}

597statusErrDef initProcessImage(void)
598{
599 tOplkError ret = kErrorOk;
600 tObdSize obdSize;
601
602 /* Allocate process image */
603 printf("Initializing process image...\n");
604 printf("Size of process image: Input = %lu Output = %lu \n",
605 (ULONG)sizeof(PI_IN),
606 (ULONG)sizeof(PI_OUT));
607 eventlog_printMessage(kEventlogLevelInfo,
608 kEventlogCategoryGeneric,
609 "Allocating process image: Input:%lu Output:%lu",
610 (ULONG)sizeof(PI_IN),
611 (ULONG)sizeof(PI_OUT));
612
613 ret = oplk_allocProcessImage(sizeof(PI_IN), sizeof(PI_OUT));
614 if (ret != kErrorOk)
615 return errOplkAllocProcessImage;
616
617 pProcessImageIn_l = (const PI_IN*)oplk_getProcessImageIn();
618 pProcessImageOut_l = (PI_OUT*)oplk_getProcessImageOut();
619
620 /* link process variables used by CN to object dictionary */
621 fprintf(stderr, "Linking process image vars:\n");
622
623 obdSize = 2;
624
625 //Link image EC of the correct NODEID
626
627 // Init process image output
628 // Example : CN3 and 3 CNs --> from nbValuesCN_Out_ByCN = 75 / 3 * (3 - 1) = 50 to nbValuesCN_Out_ByCN + nbValuesCN_Out = 50 + 25 = 75
629 for (int i = nbValuesCN_Out_ByCN; i <= nbValuesCN_Out_ByCN + nbValuesCN_Out + 1; i++)
630 {
631 if (activated_Out_CN_l[i + 1])
632 {
633 //Link valves images
634 if (i > nbValuesCN_Out_ByCN && i <= nbValuesCN_Out_ByCN + nbValuesCN_Out / 2)
635 ret = linkPDO_out(obdSize, i, 0x6500, 0x00 + (i % (nbValuesCN_Out+1)));
636 //Link sensors images
637 else if (i > nbValuesCN_Out_ByCN + nbValuesCN_Out / 2 && i <= nbValuesCN_Out_ByCN + nbValuesCN_Out)
638 ret = linkPDO_out(obdSize, i, 0x6502, 0x01 + ((i % (nbValuesCN_Out+1)) % (nbValuesCN_In+1)));
639 //Link the board state
640 else if (i == nbValuesCN_Out_ByCN)
641 ret = linkPDO_out(obdSize, i, 0x6501, NODEID);
642 if (ret != kErrorOk)
643 return errLinkPDOout;
644 }
645 }
646
647 // Init process image input
648
649 for (int i = nbValuesCN_In_ByCN; i < nbValuesCN_In_ByCN + nbValuesCN_In + 1; i++)
650 {
651 if (activated_Out_CN_l[i + nbValuesCN_In_ByCN + 2 - NODEID])
652 {
653 //Link valves images in from the MN
654 if(i > nbValuesCN_In_ByCN && i <= nbValuesCN_In_ByCN + nbValuesCN_In)
655 ret = linkPDO_in(obdSize, i, 0x6510, 0x00 + (i % (nbValuesCN_In+1)));
656 //Link the incoming general state from the MN
657 else if(i == nbValuesCN_In_ByCN)
658 ret = linkPDO_in(obdSize, nbValuesCN_In_ByCN, 0x6511, 0xF0);
659 if (ret != kErrorOk)
660 return errLinkPDOin;
661 }
662 }
663
664 fprintf(stderr, "Linking process vars... ok\n\n");
665
666 return noError;
667}
