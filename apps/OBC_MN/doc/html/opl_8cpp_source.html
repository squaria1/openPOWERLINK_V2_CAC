632tOplkError processSync(void)
633{
634 tOplkError ret;
635
636 //Wait for the OpenPOWERLINK cycle to start
637 ret = oplk_waitSyncEvent(100000);
638 if (ret != kErrorOk)
639 return ret;
640
641 //Receive OpenPOWERLINK RPDOs values
642 ret = oplk_exchangeProcessImageOut();
643 if (ret != kErrorOk)
644 return ret;
645
646 cnt_l++;
647
648 //Process PI_OUT --> variables entrant dans le MN
649 for (int i = 0; i < SIZE_OUT; i++)
650 {
651 values_In_MN_l[i] = pProcessImageOut_l->out_MN_array[i];
652 }
653
654 //Process PI_IN --> variables sortant du MN
655 for (int i = 0; i < SIZE_IN; i++)
656 {
657 if(i % (nbValuesCN_In+1) == 0)
658 pProcessImageIn_l->in_MN_array[i] = values_Out_MN_l[i];
659 }
660
661 switch (mode)
662 {
663 case automatic: // mode automatique : lecture de l'tat des vannes depuis le CSV de l'etat general actuel
664 break;
665 case manual: // mode manuel : l'tat des vannes proviennent directement du MN
666 for (int i = 1; i < SIZE_IN; i++) {
667 if (i % (nbValuesCN_In + 1) != 0)
668 pProcessImageIn_l->in_MN_array[i] = values_Out_MN_l[i];
669 }
670 break;
671 default:
672 break;
673 }
674
675 //Send OpenPOWERLINK TPDOs values
676 ret = oplk_exchangeProcessImageIn();
677
678 return ret;
679}
