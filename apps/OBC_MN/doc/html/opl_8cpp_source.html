566{
567 tOplkError ret;
568
569 ret = oplk_waitSyncEvent(100000);
570 if (ret != kErrorOk)
571 return ret;
572
573 ret = oplk_exchangeProcessImageOut();
574 if (ret != kErrorOk)
575 return ret;
576
577 cnt_l++;
578
579 int a = 0;
580 //Process PI_OUT --> variables entrant dans le MN
581
582 //Process PI_IN --> variables sortant du MN
583 for (int i = 0; i < SIZE_IN; i++)
584 {
585 if(i % (nbValuesCN_In+1) == 0)
586 pProcessImageIn_l->in_MN_array[i] = values_Out_MN_l[i];
587 }
588
589 switch (mode)
590 {
591 case automatic: // mode automatique : lecture de l'tat des vannes depuis le CSV de l'etat general actuel
592 break;
593 case manual: // mode manuel : l'tat des vannes proviennent directement du MN
594 for (int i = 1; i < SIZE_IN; i++) {
595 if (i % (nbValuesCN_In + 1) != 0)
596 pProcessImageIn_l->in_MN_array[i] = values_Out_MN_l[i];
597 }
598 break;
599 default:
600 break;
601 }
602
603 ret = oplk_exchangeProcessImageIn();
604
